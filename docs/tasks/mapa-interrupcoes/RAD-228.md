# RAD-228: Configuracao NGINX (SSL, Cache)

**Fase:** 7 - Deploy
**Tipo:** Infrastructure
**Prioridade:** Alta
**Status:** PENDENTE
**Dependencias:** RAD-213 a RAD-215

## Objetivo

Configurar NGINX como reverse proxy com SSL e cache.

## Localizacao

`deploy/nginx/`

## Implementacao

```nginx
# deploy/nginx/nginx.conf

# Upstream backends
upstream api_backend {
    server backend:8000;
    keepalive 32;
}

# Cache config
proxy_cache_path /var/cache/nginx/mapa levels=1:2 keys_zone=mapa_cache:10m max_size=1g inactive=30m use_temp_path=off;

server {
    listen 80;
    server_name portal.roraimaenergia.com.br;

    # Redirect HTTP to HTTPS
    return 301 https://$server_name$request_uri;
}

server {
    listen 443 ssl http2;
    server_name portal.roraimaenergia.com.br;

    # SSL Configuration
    ssl_certificate /etc/nginx/ssl/portal.crt;
    ssl_certificate_key /etc/nginx/ssl/portal.key;
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256;
    ssl_prefer_server_ciphers on;
    ssl_session_cache shared:SSL:10m;
    ssl_session_timeout 10m;

    # Security headers
    add_header Strict-Transport-Security "max-age=31536000; includeSubDomains" always;
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-XSS-Protection "1; mode=block" always;
    add_header Referrer-Policy "strict-origin-when-cross-origin" always;

    # Gzip
    gzip on;
    gzip_vary on;
    gzip_proxied any;
    gzip_comp_level 6;
    gzip_types text/plain text/css text/xml application/json application/javascript application/xml+rss;

    # Frontend static files
    root /var/www/mapa/dist;
    index index.html;

    location / {
        try_files $uri $uri/ /index.html;

        # Cache static assets
        location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg|woff2)$ {
            expires 1y;
            add_header Cache-Control "public, immutable";
        }
    }

    # GeoJSON - Cache longo (dados nao mudam frequentemente)
    location /geojson/ {
        alias /var/www/mapa/geojson/;
        expires 24h;
        add_header Cache-Control "public";
    }

    # API Backend
    location /api/ {
        proxy_pass http://api_backend;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_set_header Connection "";

        # Timeout configuracao
        proxy_connect_timeout 30s;
        proxy_send_timeout 60s;
        proxy_read_timeout 60s;

        # Cache para endpoints de leitura (30 min = REN 1.137)
        location ~ ^/api/mapa/(interrupcoes|estatisticas|municipios) {
            proxy_pass http://api_backend;
            proxy_cache mapa_cache;
            proxy_cache_valid 200 30m;
            proxy_cache_use_stale error timeout updating http_500 http_502 http_503 http_504;
            proxy_cache_background_update on;
            proxy_cache_lock on;

            add_header X-Cache-Status $upstream_cache_status;
        }
    }

    # Health check
    location /health {
        access_log off;
        return 200 "OK\n";
        add_header Content-Type text/plain;
    }

    # Metrics (Prometheus)
    location /metrics {
        stub_status on;
        access_log off;
        allow 10.0.0.0/8;
        deny all;
    }

    # Error pages
    error_page 500 502 503 504 /50x.html;
    location = /50x.html {
        root /var/www/error;
        internal;
    }
}
```

### Docker Compose

```yaml
# deploy/docker-compose.yml
version: "3.8"

services:
  nginx:
    image: nginx:1.25-alpine
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./nginx/nginx.conf:/etc/nginx/conf.d/default.conf:ro
      - ./nginx/ssl:/etc/nginx/ssl:ro
      - ./frontend/dist:/var/www/mapa/dist:ro
      - ./frontend/public/geojson:/var/www/mapa/geojson:ro
      - nginx_cache:/var/cache/nginx
    depends_on:
      - backend
    restart: unless-stopped
    healthcheck:
      test: ["CMD", "nginx", "-t"]
      interval: 30s
      timeout: 10s
      retries: 3

  backend:
    build: ./backend
    expose:
      - "8000"
    environment:
      - ORACLE_DSN=${ORACLE_DSN}
      - REDIS_URL=redis://redis:6379/0
    depends_on:
      - redis
    restart: unless-stopped

  redis:
    image: redis:7-alpine
    volumes:
      - redis_data:/data
    restart: unless-stopped

volumes:
  nginx_cache:
  redis_data:
```
