## Objetivo

Criar componente de grafico mostrando evolucao das interrupcoes ao longo do tempo.

`frontend/src/components/GraficoEvolucao/GraficoEvolucao.tsx`

## Implementacao

```typescript
// src/components/GraficoEvolucao/GraficoEvolucao.tsx
import { useMemo } from 'react';
import type { EstatisticasMapa } from '@/types';

interface GraficoEvolucaoProps {
  dados: Array<{
    timestamp: string;
    totalInterrupcoes: number;
    totalUcs: number;
  }>;
}

export function GraficoEvolucao({ dados }: GraficoEvolucaoProps) {
  const maxValor = useMemo(
    () => Math.max(...dados.map((d) => d.totalUcs), 1),
    [dados]
  );

  if (dados.length === 0) {
    return (
      <div className="bg-white rounded-lg shadow p-4 text-center text-gray-500">
        Sem dados de evolucao disponiveis
      </div>
    );
  }

  return (
    <div className="bg-white rounded-lg shadow p-4">
      <h3 className="font-semibold mb-4">Evolucao (ultimas 24h)</h3>

      <div className="flex items-end gap-1 h-32">
        {dados.map((item, index) => {
          const altura = (item.totalUcs / maxValor) * 100;
          const hora = new Date(item.timestamp).getHours();

          return (
            <div
              key={index}
              className="flex-1 flex flex-col items-center group"
            >
              <div
                className="w-full bg-blue-500 rounded-t transition-all hover:bg-blue-600"
                style={{ height: `${altura}%` }}
                title={`${item.totalUcs.toLocaleString()} UCs`}
              />
              <span className="text-xs text-gray-500 mt-1">{hora}h</span>

              {/* Tooltip */}
              <div className="hidden group-hover:block absolute bottom-full mb-2 bg-gray-800 text-white text-xs px-2 py-1 rounded">
                {item.totalInterrupcoes} int.
                <br />
                {item.totalUcs.toLocaleString()} UCs
              </div>
            </div>
          );
        })}
      </div>

      <div className="flex justify-between text-xs text-gray-500 mt-2">
        <span>Interrupcoes</span>
        <span>UCs Afetadas</span>
      </div>
    </div>
  );
}
```
