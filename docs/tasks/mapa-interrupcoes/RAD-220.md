```typescript
// src/components/FiltroFaixaDuracao/FiltroFaixaDuracao.tsx
import { useState } from 'react';
import type { FaixaDuracao } from '@/types';

const FAIXAS: Array<{ value: FaixaDuracao; label: string; cor: string }> = [
  { value: 'ATE_1H', label: 'Menos de 1h', cor: '#4CAF50' },
  { value: 'DE_1_A_3H', label: '1 a 3h', cor: '#8BC34A' },
  { value: 'DE_3_A_6H', label: '3 a 6h', cor: '#FFEB3B' },
  { value: 'DE_6_A_12H', label: '6 a 12h', cor: '#FF9800' },
  { value: 'DE_12_A_24H', label: '12 a 24h', cor: '#FF5722' },
  { value: 'DE_24_A_48H', label: '24 a 48h', cor: '#F44336' },
  { value: 'MAIS_48H', label: 'Mais de 48h', cor: '#B71C1C' },
];

interface FiltroFaixaDuracaoProps {
  selecionadas: FaixaDuracao[];
  onChange: (faixas: FaixaDuracao[]) => void;
}

export function FiltroFaixaDuracao({
  selecionadas,
  onChange,
}: FiltroFaixaDuracaoProps) {
  const toggleFaixa = (faixa: FaixaDuracao) => {
    if (selecionadas.includes(faixa)) {
      onChange(selecionadas.filter((f) => f !== faixa));
    } else {
      onChange([...selecionadas, faixa]);
    }
  };

  const toggleTodas = () => {
    if (selecionadas.length === FAIXAS.length) {
      onChange([]);
    } else {
      onChange(FAIXAS.map((f) => f.value));
    }
  };

  return (
    <div className="bg-white rounded-lg shadow p-4">
      <div className="flex justify-between items-center mb-3">
        <h3 className="font-semibold">Filtrar por Duracao</h3>
        <button
          className="text-sm text-blue-600 hover:underline"
          onClick={toggleTodas}
        >
          {selecionadas.length === FAIXAS.length ? 'Limpar' : 'Todas'}
        </button>
      </div>

      <div className="flex flex-wrap gap-2">
        {FAIXAS.map(({ value, label, cor }) => {
          const isSelected = selecionadas.includes(value);
          return (
            <button
              key={value}
              className={`
                px-3 py-1.5 rounded-full text-sm font-medium
                transition-all duration-200
                ${isSelected
                  ? 'text-white shadow-md'
                  : 'bg-gray-100 text-gray-600 hover:bg-gray-200'
                }
              `}
              style={{
                backgroundColor: isSelected ? cor : undefined,
              }}
              onClick={() => toggleFaixa(value)}
            >
              {label}
            </button>
          );
        })}
      </div>
    </div>
  );
}
```
